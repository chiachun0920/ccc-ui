Dock：時尚化的邊欄

說明

        
    移動你的滑鼠到視窗的最下面，你會看到...

介面

        
    @Input() icon: Icon;
    @Output() iconClick: EventEmitter<Icon>

Icon結構

        
    export class Icon {
        public iconImg: string;
        public title: string;
    }

使用方法


    <ccc-dock 
        [icon]="appIcon" 
        (iconClick)="onClick($event)"></ccc-dock>
    

    public appIcon = [
        { iconImg: '', title: 'panelBar', path: 'panel-bar' }
    ];
