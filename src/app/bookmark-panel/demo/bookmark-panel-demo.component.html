Bookmark-Panel：書籤面板

說明


    透過Component動態載入，來抽換左右兩側不同的Component

介面


        @Input() lTabs: PanelBarTab[];
        @Input() rTabs: PanelBarTab[];

PanelBarTab結構


    export class Icon {
        public component: Type<any>;
        public name: string;
        public icon: string;
    }

使用方法


    <ccc-panel-bar [lTabs]="lTabs" [rTabs]="rTabs"></ccc-panel-bar>
    

    this.rTabs = [
        { component: AComponent, name: '右邊欄一', icon: 'fa fa-medkit' },
        { component: BComponent, name: '右邊欄二', icon: 'fa fa-user-md' },
        { component: CComponent, name: '右邊欄三', icon: 'fa fa-ambulance' },
    ]
    this.lTabs = [
        { component: DComponent, name: '左邊欄一', icon: 'fa fa-medkit' },
        { component: EComponent, name: '左邊欄二', icon: 'fa fa-user-md' },
        { component: FComponent, name: '左邊欄三', icon: 'fa fa-ambulance' },
    ]
