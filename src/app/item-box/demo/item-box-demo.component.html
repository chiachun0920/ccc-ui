Item-Box：

介面


    @Input() keyName: string;
    @Input() value: string;
    @Input() height: string;
    @Input() itemBoxPairs:ItemBoxPairs[];
    @Output() pairFocus: EventEmitter<ItemBoxPair>;

ItemBoxPair結構


    export class Pair {
        public key: string;
        public value: string;
    }

使用方法


    <ccc-item-box 
        [keyName]="keyName" 
        [valueName]="valueName" 
        [height]="height" 
        [itemBoxPairs]="itemBoxPairs" 
        (pairFocus)="onPairFocus($event)">
    </ccc-item-box>
